8. How many distinct years of data are in the CSV ﬁle?

SELECT COUNT(DISTINCT year(call_date)) AS distinct_years
FROM fire_data;

9. What week of the year in 2018 had the most ﬁre calls?

SELECT 
    weekofyear(call_date) AS week,
    COUNT(*) AS total_calls
FROM fire_data
WHERE year(call_date) = 2018
GROUP BY weekofyear(call_date)
ORDER BY total_calls DESC
LIMIT 1;

10. What neighborhoods in San Francisco had the worst response time in 2018?

SELECT 
    neighborhoods,
    AVG(UNIX_TIMESTAMP(response_dttm) - UNIX_TIMESTAMP(dispatch_dttm)) AS avg_response_time_sec
FROM fire_data
WHERE YEAR(call_date) = 2018
  AND response_dttm IS NOT NULL
  AND dispatch_dttm IS NOT NULL
GROUP BY neighborhoods
ORDER BY avg_response_time_sec DESC
LIMIT 10;
